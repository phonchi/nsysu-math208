def prim(graph, start): for v in graph: v.distance = ∞ v.previous = None start.distance = 0 pq = PriorityQueue(graph) while not pq.is_empty(): d, u = pq.delete() for v in u.get_neighbors(): new_d = u.weight(v) # 注意：只看單邊權！ if v in pq and new_d < v.distance: v.previous = u v.distance = new_d pq.update(v, new_d)
