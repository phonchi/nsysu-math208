insertionSort(aList):    for i in range(1, aList.size()):        curVal = aList[i]        curPos = i        while curPos > 0 && aList[curPos-1] > curVal:            aList[curPos] = aList[curPos-1]            curPos = curPos - 1        aList[curPos] = curVal
