def merge_sort(a):    if len(a) <= 1: return    mid = len(a) // 2    L, R = a[:mid], a[mid:]    merge_sort(L)    merge_sort(R)    # merge L and R back into a    i = j = k = 0    while i < len(L) and j < len(R):        if L[i] <= R[j]:            a[k] = L[i]; i += 1        else:            a[k] = R[j]; j += 1        k += 1
