def shell_sort(a):    gap = len(a) // 2    while gap > 0:        for i in range(gap, len(a)):            key, pos = a[i], i            while pos >= gap and a[pos-gap] > key:                a[pos] = a[pos-gap]                pos -= gap            a[pos] = key        gap //= 2
